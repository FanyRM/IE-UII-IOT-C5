CREATE TABLE registro_temperaturas (id SERIAL PRIMARY KEY, fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP, temp_ambiente REAL, temp_objeto REAL);

CREATE TABLE registro_densidad (id SERIAL PRIMARY KEY, fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,densidad REAL, estado TEXT);

CREATE TABLE registro_movimiento (id SERIAL PRIMARY KEY, fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP, estado TEXT);

CREATE TABLE registro_alertas (id SERIAL PRIMARY KEY, fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP, tipo TEXT, severidad TEXT);

CREATE TABLE IF NOT EXISTS encargados (id INTEGER PRIMARY KEY AUTOINCREMENT, nombre TEXT NOT NULL, correo TEXT NOT NULL UNIQUE, fecha_registro TEXT DEFAULT CURRENT_TIMESTAMP);
